git init --> INICIA GIT EN MI CARPETA DE ARCHIVOS

git add . --> PARA INDICAR LOS ARCHIVOS QUE DESEO SE SUBAN (PUEDE SER INDICANDO
EL NOMBRE DEL ARCHIVO O EL . SELECCIONA TODOS LOS ARCHIVOS)

git commit -m "first commit" --> PARA ESCRIBIR EL MENSAJE QUE ACOMPAÑA A MI CÓDIGO

git remote add origin https://github.com/NOMBRE_USUARIO/NOMBRE_PROYECTO.git --> PARA 
INDICAR A QUE REPOSITORIO VA

git push -u origin master --> PARA INDICAR A QUE RAMA DEL REPOSITORIO VA
